const [N, A, M, B] = input.map((v) => v.split(' ').map(Number));

    let mid = Math.floor((left + right) / 2);

    while (left <= right) {
        if (el === A[mid]) {
            ans.push(1);
            return;
        } else if (el > A[mid]) {
            left = mid + 1;
        } else {
            right = mid - 1;
        }

        mid = Math.floor((left + right) / 2);
    }

    ans.push(0);
    return;
});

console.log(ans.join('\n'));
